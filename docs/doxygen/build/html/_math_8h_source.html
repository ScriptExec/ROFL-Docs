340 constexpr Numeric EaseOut(Numeric time, Numeric startV, Numeric change, float duration)
341 {
342 time /= duration / 2;
343 if (time < 1)
344 {
345 return change / 2 * time * time + startV;
346 }
347
348 --time;
349 return -change / 2 * (time * (time - 2) - 1) + startV;
