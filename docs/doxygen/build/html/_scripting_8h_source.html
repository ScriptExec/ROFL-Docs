7 #define Function(f) asFunctionPtr(f)
8 #if (defined(_MSC_VER) && _MSC_VER <= 1200) || (defined(__BORLANDC__) && __BORLANDC__ < 0x590)
9 // MSVC 6 has a bug that prevents it from properly compiling using the correct asFUNCTIONPR with operator >
10 // so we need to use ordinary C style cast instead of static_cast. The drawback is that the compiler can't
11 // check that the cast is really valid.
12 // BCC v5.8 (C++Builder 2006) and earlier have a similar bug which forces us to fall back to a C-style cast.
13 #define FunctionPr(f,p,r) asFunctionPtr((void (*)())((r (*)p)(f)))
14 #else
15 #define FunctionPr(f,p,r) asFunctionPtr(reinterpret_cast<void (*)()>(static_cast<r (*)p>(f)))
16 #endif
